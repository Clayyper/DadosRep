DEF TEMP-TABLE tt-itens
    FIELD cod_grp_usuar   LIKE prog_dtsul_segur.cod_grp_usuar
    FIELD cod_prog_dtsul  LIKE prog_dtsul_segur.cod_prog_dtsul.

DEF VAR d-aux as decimal INIT 0.

DEF VAR c-arquivo AS CHAR FORMAT "x(50)".
UPDATE c-arquivo.

MESSAGE "SEARCH(c-arquivo): " SEARCH(c-arquivo) VIEW-AS ALERT-BOX.

INPUT FROM VALUE(SEARCH(c-arquivo)).
REPEAT :
     CREATE tt-itens.
     IMPORT DELIMITER ";" tt-itens.
END.
INPUT CLOSE.

MESSAGE "CAN-FIND(FIRST tt-itens): " CAN-FIND(FIRST tt-itens) VIEW-AS ALERT-BOX.

FOR EACH prog_dtsul_segur:
    IF prog_dtsul_segur.idi_dtsul > d-aux THEN
        ASSIGN d-aux = prog_dtsul_segur.idi_dtsul.
END.

FOR EACH tt-itens.

    FIND prog_dtsul_segur WHERE prog_dtsul_segur.cod_grp_usuar=tt-itens.cod_grp_usuar
                            AND prog_dtsul_segur.cod_prog_dtsul=tt-itens.cod_prog_dtsul NO-ERROR.
    IF AVAIL prog_dtsul_segur THEN NEXT.
    
    FIND grp_usuar WHERE cod_grp_usuar = tt-itens.cod_grp_usuar NO-ERROR.
        IF AVAIL grp_usuar THEN DO:
            FIND prog_dtsul WHERE cod_prog_dtsul = tt-itens.cod_prog_dtsul NO-ERROR.
            IF AVAIL prog_dtsul THEN DO:
                MESSAGE tt-itens.cod_grp_usuar tt-itens.cod_prog_dtsul grp_usuar.idi_dtsul prog_dtsul.idi_dtsul  VIEW-AS ALERT-BOX.
                    CREATE prog_dtsul_segur.
                    ASSIGN      prog_dtsul_segur.cod_grp_usuar=tt-itens.cod_grp_usuar
                                prog_dtsul_segur.cod_prog_dtsul=tt-itens.cod_prog_dtsul
                                prog_dtsul_segur.idi_dtsul_grp_usuar=grp_usuar.idi_dtsul
                                prog_dtsul_segur.idi_dtsul_prog_dtsul=prog_dtsul.idi_dtsul
                                prog_dtsul_segur.idi_dtsul = d-aux + 1 .
                    ASSIGN d-aux = d-aux + 1.



            END.
        END.


END.
