Requisitante nao cadastrados

OUTPUT TO VALUE('C:\TEMP\Requisitantes.TXT').
FOR EACH ordem-compra WHERE numero-ordem EQ 71000:
    FIND usuar_mestre WHERE cod_usuario EQ ordem-compra.requisitante NO-LOCK NO-ERROR.
    IF NOT AVAIL usuar_mestre THEN DO:
        PUT UNFORMATTED ordem-compra.numero-ordem ";" ordem-compra.dat-ordem ";" ordem-compra.requisitante SKIP.
        DISP ordem-compra.requisitante WITH 1 COLUMN WIDTH 320.
    END.
END.
OUTPUT CLOSE.

Ordem status recebida porem com saldo de fornecedor.

OUTPUT TO VALUE('C:\TEMP\CompradorRequisitanteRecebida.TXT').
PUT UNFORMATTED "PEDIDO" ";" "ORDEM"  ";" "COMRPADOR" ";" "REQUISITANTE" ";" "DATA ORDEM" ";" "CONTA CONTABIL" ";" "CENTRO CUSTO" SKIP.
FOR EACH prazo-compr WHERE (prazo-compr.situacao EQ 6)
                        AND qtd-sal-forn > 0:

    FIND ordem-compra OF prazo-compr WHERE dat-ordem >= 01/01/2002 
                                        AND dat-ordem <= 12/31/2017 
                                        NO-ERROR.
    IF AVAIL ordem-compra THEN DO:
        FIND pedido-compr OF ordem-compra NO-ERROR.
        FIND usuar_mestre WHERE cod_usuario EQ ordem-compra.cod-comprado NO-LOCK NO-ERROR.
        IF NOT AVAIL usuar_mestre THEN DO:
            PUT UNFORMATTED pedido-compr.num-pedido ";" prazo-compr.numero-ordem  ";" ordem-compra.cod-comprado ";" ordem-compra.requisitante ";" ordem-compra.dat-ordem ";" ordem-compra.ct-codigo ";" ordem-compra.sc-codigo SKIP.
            ASSIGN ordem-compra.cod-comprado = "eula.g".
        END.
        FIND usuar_mestre WHERE cod_usuario EQ ordem-compra.requisitante NO-LOCK NO-ERROR.
        IF NOT AVAIL usuar_mestre THEN DO:
            PUT UNFORMATTED pedido-compr.num-pedido ";" prazo-compr.numero-ordem  ";" ordem-compra.cod-comprado ";" ordem-compra.requisitante ";" ordem-compra.dat-ordem ";" ordem-compra.ct-codigo ";" ordem-compra.sc-codigo SKIP.
            ASSIGN ordem-compra.requisitante = "eula.g".
        END.
        IF ordem-compra.ct-codigo EQ "" THEN DO:
            PUT UNFORMATTED pedido-compr.num-pedido ";" prazo-compr.numero-ordem  ";" ordem-compra.cod-comprado ";" ordem-compra.requisitante ";" ordem-compra.dat-ordem ";" ordem-compra.ct-codigo ";" ordem-compra.sc-codigo SKIP.
            ASSIGN ct-codigo ="63180000"
                   sc-codigo = "11070".
        END.
        
    END.

END.
OUTPUT CLOSE.


CORRIGINDO ERRO DE DEPOSITO INEXISTENTE

OUTPUT TO VALUE('C:\TEMP\DEPOSITO.TXT').
PUT UNFORMATTED "PEDIDO" ";" "ORDEM"  ";" "COMRPADOR" ";" "REQUISITANTE" ";" "DATA ORDEM" ";" "CONTA CONTABIL" ";" "CENTRO CUSTO" SKIP.
FOR EACH prazo-compr WHERE (prazo-compr.situacao NE 4)
                        AND qtd-sal-forn > 0:

    FIND ordem-compra OF prazo-compr WHERE dat-ordem >= 01/01/2002 
                                        AND dat-ordem <= 12/31/2017 
                                        NO-ERROR.
    IF AVAIL ordem-compra THEN DO:
        FIND pedido-compr OF ordem-compra NO-ERROR.
        FIND usuar_mestre WHERE cod_usuario EQ ordem-compra.cod-comprado NO-LOCK NO-ERROR.
        
        IF ordem-compra.dep-almoxar EQ "" THEN DO:
            PUT UNFORMATTED pedido-compr.num-pedido ";" prazo-compr.numero-ordem  ";" ordem-compra.cod-comprado ";" ordem-compra.requisitante ";" ordem-compra.dat-ordem ";" ordem-compra.dep-almoxar SKIP.
            ASSIGN ordem-compra.dep-almoxar ="ALM".
        
        END.
        
    END.

END.
OUTPUT CLOSE.
