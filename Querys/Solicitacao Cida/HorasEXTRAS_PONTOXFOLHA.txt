      ,CAST(quantidade as numeric(18,3))/60
  FROM [LEP].[dbo].[HORASEXTRAS2]
GO


update a
set a.matricula = b.[Mat_Ant]
from (SELECT * FROM dbo.HORASEXTRAS2) as a
inner join dbo.HORASEXTRAS2 as b on b.indice = a.indice -1 and a.Matricula = 0


update a
set a.matricula = b.Matricula
from (SELECT * FROM dbo.HORASEXTRAS2 where Matricula is null) as a
inner join dbo.HORASEXTRAS2 as b on b.indice = a.indice -1 


update dbo.HORASEXTRAS2

update dbo.HORASEXTRAS2
set quantidade = CAST(LTRIM(DATEDIFF(MINUTE, 0, [horaextra])) AS INT)

update dbo.HORASEXTRAS2
set convertido = CAST(quantidade as numeric(18,3))/60
